To print all character counts in an array

public static void getAllCharCount(String name){

HashMap<Character,Integer> hm = new hashMap<>();
char[] c = hm.toCharArray();

for(char ch : c){
if(hm.containsKey(ch)){
hm.put(ch, hm.get(ch)+1);
}else{
hm.put(ch, 1);
}
}
}
